Column A Column B

Code

<table>
  <caption>Table Caption</caption>
  <thead>
    <tr>
      <th>Column A</th>
      <th>Column B</th>
    </tr>
  </thead>
  <tbody>
    <tr>
      <td>Data Row 1, Column A</td>
      <td>Data Row 1, Column B</td>
    </tr>
    <tr>
      <td>Data Row 2, Column A</td>
      <td>Data Row 2, Column B</td>
    </tr>
  </tbody>
  <tfoot>
    <tr>
      <td>Foot of Column A</td>
      <td>Foot of Column B</td>
    </tr>
  </tfoot>
</table>
